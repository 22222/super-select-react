# Chakra UI

This example shows one way to define a Chakra-flavored wrapper around `SuperSelect`.

```tsx
export function ChakraSuperSelect(props) {
    return (
        <SuperSelect
            {...props}
            customization={{
                classNamePrefix: "chakra-super-select",
                modalSelectButton: {
                    component: ChakraSelectButton,
                    selectedContent: {
                        component: ChakraSelectedContent,
                        placeholder: "\u200B",
                    },
                },
                modal: {
                    component: ChakraModalDialog,
                    okButton: {
                        component: ChakraOkButton,
                    },
                },
                searchInput: {
                    component: ChakraSearchInput,
                    placeholder: "Search options",
                },
                optionListInput: {
                    component: ChakraOptionListInput,
                },
                toggleButtonInput: {
                    component: ChakraToggleButtonInput,
                },
            }}
        />
    );
}
```
